## Install

```bash
npm install --save react-dater
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-dater'
import 'react-dater/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
